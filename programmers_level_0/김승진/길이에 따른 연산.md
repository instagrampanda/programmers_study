```c++
#include <string>
#include <vector>

using namespace std;

int solution(vector<int> num_list) {
    int answer = 0;
    int len = num_list.size();
    if (len > 10)
    {
        for (int i = 0; i < len; i++)
        {
            answer += num_list[i];
        }
    }
    else
    {
        answer = 1;
        for (int i = 0; i < len; i++)
        {
            answer *= num_list[i];
        }
    }
    return answer;
}
```
