class Solution {
    public int solution(int[] array) {
        int[] frequency = new int[1001]; // 배열의 크기는 문제에서 주어진 범위에 따라 설정
        int maxFrequency = 0;
        int maxNum = -1;

        for (int num : array) {
            frequency[num]++;
            if (frequency[num] > maxFrequency) {
                maxFrequency = frequency[num];
                maxNum = num;
            } else if (frequency[num] == maxFrequency) {
                maxNum = -1; // 최빈값이 중복될 때 -1 반환
            }
        }

        return maxNum;
    }
}
